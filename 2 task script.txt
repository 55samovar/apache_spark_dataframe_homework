from pyspark.sql.functions import col
covid_df = spark.read.option('header', True).option('sep', ',').option('inferSchema', True).csv('covid-data.csv')
covid_df.select('date', 'location', 'new_cases').where((col('date') >= '2021-03-22') & (col('date') <= '2021-03-28')).groupBy('location').max('new_cases').orderBy(col('max(new_cases)').desc()).show()
